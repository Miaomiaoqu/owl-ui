      <li ref="tab"
          v-for="(item, index) in data"
          :key="index"
          :class="index === activeIndex ? 'active' : ''"
          :style="index === activeIndex ? [activeColor, activeStyle] : defaultColor"

        {{name ? item[name] : item}}
