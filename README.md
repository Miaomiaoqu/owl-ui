### Install

```bash
npm install owl-ui -S
```

### Usage

```js
import Vue from 'vue'
import Owl from 'owl-ui'

Vue.use(Owl)
```

### Development

```bash
git clone git@github.com:dengwb1991/owl-ui.git

cd owl-ui

npm install

npm run dev

# or run document development

npm run docs
```
